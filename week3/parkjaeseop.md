```py
#Example1

dan = int(input("1~9까지의 숫자를 입력하세요. : "))

if 1<= dan <= 9 :
    for i in range(1,10) :
        print("%2d X %2d = %2d" %(dan, i, dan*i))

else :
    print("1에서 9까지의 수를 다시 입력하세요:")
```
